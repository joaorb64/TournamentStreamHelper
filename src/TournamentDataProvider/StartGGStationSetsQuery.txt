query StationSetsQuery($eventSlug: String!, $filters: SetFilters = {}) {
  event(slug: $eventSlug) {
    sets(page: 1, perPage: 999, filters: $filters) {
      nodes {
        id
        state
        slots {
          entrant {
            id
          }
        }
        station {
          id
        }
      }
    }
  }
}
